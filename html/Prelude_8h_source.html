70 // builtin functions templated by a function type. A function
71 // of such type, when applied, returns an attribute
72 template <typename ATTRT> struct BuiltinOp : public Op {
73  typedef const Attr *(*attrT)();
74  attrT func;
75  BuiltinOp(ATTRT func) : Op(Builtin), func((attrT)func) {}
76  static bool classof(const Op *op) { return op->getOpType() == Builtin; }
77 };
78 
79 // inlined functions templated by a function type. A function
80 // of such type, when applied, returns an expression as the function body
81 template <typename EXPRT> struct InlinedOp : public Op {
82  typedef const Expr *(*exprT)();
83  exprT func;
84  InlinedOp(EXPRT func) : Op(Inlined), func((exprT)func) {}
85  static bool classof(const Op *op) { return op->getOpType() == Inlined; }
86 };
