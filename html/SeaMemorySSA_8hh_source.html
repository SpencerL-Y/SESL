419  // After deleting incoming position I, the order of incoming may be changed.
420  void unorderedDeleteIncoming(unsigned I) {
421  unsigned E = getNumOperands();
422  assert(I < E && "Cannot remove out of bounds Phi entry.");
423  // MemoryPhi must have at least two incoming values, otherwise the MemoryPhi
424  // itself should be deleted.

434  // After deleting entries that satisfy Pred, remaining entries may have
